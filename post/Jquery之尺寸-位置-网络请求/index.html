    headers:{},// 请求添加的额外头信息
    data:{}/string, // 请求体数据
    dataType:’json/jsonp’,// 服务器返回数据类型
    success:function(ret){},// 请求成功回调
    error:function(){},// 请求失败回调
})

Promise.all()、Promise.race()

race：只要有一个执行完成，不管执行成功或者失败，程序结束；all：等待所有的promise完成，程序才结束，并且所有的promise一个都不能有错。
